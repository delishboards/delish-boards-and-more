    // Example: wire the button to the cart (main.js handles add-to-cart buttons site-wide)
    document.getElementById('add-single')?.addEventListener('click', () => {
      // create a temporary product object and add to cart via same localStorage schema
      const product = { id: 'serving-knife', title: document.getElementById('product-title').textContent, price: 12.00 };
      const cart = JSON.parse(localStorage.getItem('delish_cart') || '[]');
      const existing = cart.find(i => i.id === product.id);
      if (existing) existing.qty += 1;
      else cart.push({...product, qty:1});
      localStorage.setItem('delish_cart', JSON.stringify(cart));
      alert('Added to cart');
      location.href = 'shop.html';
    });
